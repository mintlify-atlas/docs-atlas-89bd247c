---
title: Delete a redirect
description: Permanently delete an existing redirect
---

## Endpoint

```
DELETE /delete-redirect/{id}
```

## Function signature

```typescript
export const deleteRedirect = async (id: string): Promise<AxiosResponse>
```

## Parameters

<ParamField path="id" type="string" required>
  The ID of the redirect to be deleted
</ParamField>

## Response

<ResponseField name="status" type="number">
  HTTP status code (typically 200 or 204 on success)
</ResponseField>

<ResponseField name="data" type="object">
  Confirmation object containing deletion details
</ResponseField>

## Code example

<CodeGroup>

```typescript Example
import { deleteRedirect } from '@/lib/api';

const removeRedirect = async (redirectId: string) => {
  const response = await deleteRedirect(redirectId);
  console.log('Redirect deleted successfully');
  return response.data;
};
```

```typescript With confirmation
import { deleteRedirect } from '@/lib/api';

const removeRedirectWithConfirmation = async (redirectId: string) => {
  const confirmed = window.confirm(
    'Are you sure you want to delete this redirect? This action cannot be undone.'
  );
  
  if (!confirmed) {
    return null;
  }
  
  try {
    const response = await deleteRedirect(redirectId);
    return response.data;
  } catch (error) {
    console.error('Failed to delete redirect:', error);
    throw error;
  }
};
```

```typescript Delete multiple redirects
import { deleteRedirect } from '@/lib/api';

const deleteMultipleRedirects = async (redirectIds: string[]) => {
  const deletionPromises = redirectIds.map(id => deleteRedirect(id));
  
  try {
    const results = await Promise.all(deletionPromises);
    console.log(`Successfully deleted ${results.length} redirects`);
    return results;
  } catch (error) {
    console.error('Failed to delete some redirects:', error);
    throw error;
  }
};
```

</CodeGroup>

<Note>
  Deleting a redirect is permanent and cannot be undone. All analytics data associated with the redirect will also be deleted.
</Note>

## Error responses

<Expandable title="401 - Unauthorized">
  Returned when the user is not authenticated
</Expandable>

<Expandable title="403 - Forbidden">
  Returned when the user does not have permission to delete this redirect
</Expandable>

<Expandable title="404 - Not Found">
  Returned when the redirect with the specified ID does not exist
</Expandable>

<Expandable title="500 - Internal Server Error">
  Returned when an unexpected error occurs on the server
</Expandable>
